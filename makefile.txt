# Compiler and flags
CXX = g++
CXXFLAGS = -std=c++17 -Wall -Wextra -O2
# Added -MMD to generate .d dependency files automatically
CPPFLAGS = -MMD -MP 

# The name of your final executable
TARGET = rpg_game

# Source and Object files
SRCS = demo.cpp Hero.cpp RPGEngine.cpp Task.cpp TaskManager.cpp
OBJS = $(SRCS:.cpp=.o)
DEPS = $(SRCS:.cpp=.d)

# Default target
all: $(TARGET)

# Link the executable
$(TARGET): $(OBJS)
	$(CXX) $(CXXFLAGS) $(OBJS) -o $(TARGET)

# Compile individual .cpp files into .o files
# Uses automatic variables: $< is the source file, $@ is the target .o
%.o: %.cpp
	$(CXX) $(CXXFLAGS) $(CPPFLAGS) -c $< -o $@

# Include the generated dependency files (.d)
# This handles header changes intelligently
-include $(DEPS)

# Run the program
run: $(TARGET)
	./$(TARGET)

# Clean generated files
clean:
	rm -f *.o *.d $(TARGET)

# Rebuild from scratch
rebuild: clean all

.PHONY: all run clean rebuild
